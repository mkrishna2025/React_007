// Creating Clas
Before ES6

	return {}
} 
ES6
class Calculator {
}

// Class with 2 public methods
var Calculator = function(){
	return {
		add: function(x, y) {
			return x + y;
		},
		sub: function(x, y){
			return x - y;
		}
	}
}
ES6
class Calculator {
	add(x, y) {
		return x + y;
	}
	sub(x, y) {
		return x - y;
	}
}
